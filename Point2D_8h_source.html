134  int compare(Point2D const& rhs) const
135  {
136  double alpha = 0.1; //minimum value to consider points different
137  double tmpx,tmpy;
138 
139  tmpx = _x - rhs._x;
140  tmpy = _y - rhs._y;
141 
142 
143  if ( (fabs(tmpx) < alpha) && (fabs(tmpy) < alpha) ) return 0;
144 
145  //else
146 
147  double mod_rhs = rhs._x*rhs._x + rhs._y*rhs._y;
148  double mod_this = _x*_x + _y*_y;
149 
150  double mod_diff = mod_rhs - mod_this;
151 
152  if (mod_diff > 0) return 1;
153  else return -1;
154  }

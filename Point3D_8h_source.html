146  bool compare(Point3D const& rhs) const
147  {
148  double tmpx,tmpy,tmpz;
149  double alpha = 0.1; //minimum value to consider points different
150 
151  tmpx = _x - rhs._x;
152  tmpy = _y - rhs._x;
153  tmpz = _z - rhs._x;
154 
155  if ((fabs(tmpx) < alpha) && (fabs(tmpy) < alpha) && (fabs(tmpz) < alpha))
156  return true;
157  else
158  return false;
159  }

173  inline bool operator < (Point3D const& rhs) const{
174  if (compare(rhs)) return false;
175 
176  double mod1, mod2;
177 
178  mod1 = sqrt(_x*_x + _y*_y + _y*_y);
179  mod2 = sqrt(rhs.getX()*rhs.getX() + rhs.getY()*rhs.getY() + rhs.getZ()*rhs.getZ());
180 
181  return (mod1 < mod2);
182 
183  }

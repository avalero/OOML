55 Component operator+(Component const& lhs, Component const& rhs)
56 {
57  CompositeComponent result(Union::create());
58 
59  //Keep all the links of the children
60  //avoid link 0, which is the default link at the origin
61  if(lhs.hasLinks())
62  for (int i=1; i<lhs.getLinks().size();i++)
63  result.addLink(lhs.getLink(i));
64 
65  if (rhs.hasLinks())
66  for (int i=1; i<rhs.getLinks().size();i++)
67  result.addLink(rhs.getLink(i));
68 
69 
70  result << lhs.get() << rhs.get();
71  return result;
72 }
